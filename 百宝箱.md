// app/middleware/auth.ts
import { NextRequest, NextResponse } from 'next/server'
import jwt from 'jsonwebtoken'

export interface AuthUser {
  id: string
  phone?: string
  membershipType: string
}

export async function requireAuth(req: NextRequest): Promise<AuthUser | null> {
  const token = req.headers.get('authorization')?.replace('Bearer ', '')
  
  if (!token) {
    return null
  }
  
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET!) as AuthUser
    return decoded
  } catch (error) {
    return null
  }
}

export async function withAuth(
  req: NextRequest,
  handler: (user: AuthUser) => Promise<NextResponse>
) {
  const user = await requireAuth(req)
  
  if (!user) {
    return NextResponse.json(
      { error: '请先登录', code: 'UNAUTHORIZED' },
      { status: 401 }
    )
  }
  
  return handler(user)
}
会员校验中间件

```typescript
// app/middleware/membership.ts
import { NextRequest, NextResponse } from 'next/server'
import { requireAuth, AuthUser } from './auth'
import { Env, generateId } from '@/lib/db'

export interface MembershipCheckResult {
  allowed: boolean
  user: AuthUser
  needMembership?: boolean
  expired?: boolean
}

export async function requireMembership(
  req: NextRequest,
  env: Env,
  requiredLevel: 'basic' | 'premium' = 'premium'
): Promise<MembershipCheckResult | NextResponse> {
  // 1. 认证检查
  const user = await requireAuth(req)
  if (!user) {
    return NextResponse.json(
      { error: '请先登录', code: 'UNAUTHORIZED' },
      { status: 401 }
    )
  }
  
  // 2. 从数据库获取最新会员信息
  const dbUser = await env.DB.prepare(
    'SELECT membership_type, membership_expire_at FROM users WHERE id = ?'
  ).bind(user.id).first()
  
  // 3. 免费用户拦截
  if (dbUser?.membership_type === 'free') {
    return NextResponse.json(
      { 
        error: '此功能需要开通会员',
        code: 'NEED_MEMBERSHIP',
        needMembership: true 
      },
      { status: 403 }
    )
  }
  
  // 4. 会员过期检查
  if (dbUser?.membership_expire_at && new Date() > new Date(dbUser.membership_expire_at)) {
    return NextResponse.json(
      { 
        error: '会员已过期，请续费',
        code: 'MEMBERSHIP_EXPIRED',
        expired: true 
      },
      { status: 403 }
    )
  }
  
  // 5. 记录访问日志
  await env.DB.prepare(
    'INSERT INTO membership_logs (id, user_id, action, feature_name, result, created_at) VALUES (?, ?, ?, ?, ?, ?)'
  ).bind(
    generateId(),
    user.id,
    'analyze',
    requiredLevel === 'premium' ? 'advanced_stats' : 'basic',
    'allowed',
    new Date().toISOString()
  ).run()
  
  return { allowed: true, user }
}
```

```typescript
// app/api/kuaile8/analyze/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { withAuth } from '@/middleware/auth'
import { requireMembership } from '@/middleware/membership'
import { Env } from '@/lib/db'

// Cloudflare Pages Functions 格式
export const onRequest: PagesFunction<Env> = async (context) => {
  const { request, env } = context
  
  return withAuth(request as NextRequest, async (user) => {
    // 会员校验
    const membershipCheck = await requireMembership(request as NextRequest, env, 'premium')
    if (membershipCheck instanceof NextResponse) {
      return membershipCheck
    }
    
    const { analyzeType, params } = await request.json()
    
    // 获取历史数据
    const historyDays = membershipCheck.user.membershipType === 'free' ? 30 : 365
    const startDate = new Date()
    startDate.setDate(startDate.getDate() - historyDays)
    
    const { results } = await env.DB.prepare(
      'SELECT * FROM lottery_results WHERE lottery_type = ? AND draw_date >= ? ORDER BY draw_date DESC'
    ).bind('kuaile8', startDate.toISOString()).all()
    
    // 解析 numbers 字段（存储为 JSON 字符串）
    const historyData = results.map(row => ({
      ...row,
      numbers: JSON.parse(row.numbers as string)
    }))
    
    // 执行分析
    let result
    if (analyzeType === 'advanced') {
      result = await advancedAnalysis(historyData, params)
    } else {
      result = await basicAnalysis(historyData, params)
    }
    
    // 更新使用次数
    await env.DB.prepare(
      'UPDATE users SET today_analyze_count = today_analyze_count + 1, total_analyze_count = total_analyze_count + 1, updated_at = ? WHERE id = ?'
    ).bind(new Date().toISOString(), user.id).run()
    
    return NextResponse.json(result)
  })
}

async function basicAnalysis(data: any[], params: any) {

    frequency: calculateFrequency(data),
    aiPrediction: await getAIPrediction(data),
    trendAnalysis: getTrendAnalysis(data),
    recommendation: getRecommendation(data)
  }
}
```

// src/utils/request.ts
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { useAuthStore } from '@/stores/auth'
import { ElMessage } from 'element-plus'

class Request {
  private instance: AxiosInstance
  
  constructor() {
    this.instance = axios.create({
      baseURL: import.meta.env.VITE_API_BASE_URL,
      timeout: 30000,
      headers: {
        'Content-Type': 'application/json'
      }
    })
    
    this.setupInterceptors()
  }
  
  private setupInterceptors() {
    // 请求拦截器
    this.instance.interceptors.request.use(
      (config) => {
        const authStore = useAuthStore()
        if (authStore.token) {
          config.headers.Authorization = `Bearer ${authStore.token}`
        }
        return config
      },
      (error) => {
        return Promise.reject(error)
      }
    )
    
    // 响应拦截器
    this.instance.interceptors.response.use(
      (response: AxiosResponse) => {
        return response.data
      },
      (error) => {
        if (error.response) {
          const { status, data } = error.response
          
          switch (status) {
            case 401:
              ElMessage.error('请先登录')
              // 跳转到登录页
              break
            case 403:
              if (data.code === 'NEED_MEMBERSHIP') {
                ElMessage.warning('此功能需要开通会员')
                // 弹出会员升级弹窗
              } else if (data.code === 'MEMBERSHIP_EXPIRED') {
                ElMessage.warning('会员已过期，请续费')
              } else {
                ElMessage.error(data.error || '权限不足')
              }
              break
            case 429:
              ElMessage.warning('请求过于频繁，请稍后再试')
              break
            default:
              ElMessage.error(data.error || '请求失败')
          }
        } else {
          ElMessage.error('网络连接失败')
        }
        return Promise.reject(error)
      }
    )
  }
  
  get<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.get(url, config)
  }
  
  post<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.post(url, data, config)
  }
  
  put<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.put(url, data, config)
  }
  
  delete<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.delete(url, config)
  }
}

export default new Request()
4.4 会员状态管理

// src/stores/membership.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import request from '@/utils/request'

interface MembershipBenefits {
  daily_analyze_limit: number
  advanced_stats: boolean
  ai_predict: boolean
  export_data: boolean
  history_range_days: number
}

interface MembershipStatus {
  is_member: boolean
  membership_type: string
  expire_at: string | null
  days_left: number
  today_used: number
  benefits: MembershipBenefits
}

export const useMembershipStore = defineStore('membership', () => {
  const status = ref<MembershipStatus | null>(null)
  const loading = ref(false)
  
  const isMember = computed(() => status.value?.is_member || false)
  const membershipType = computed(() => status.value?.membership_type || 'free')
  const daysLeft = computed(() => status.value?.days_left || 0)
  const benefits = computed(() => status.value?.benefits)
  
  // 获取会员状态
  const fetchStatus = async () => {
    loading.value = true
    try {
      const data = await request.get('/api/membership/status')
      status.value = data
    } catch (error) {
      console.error('获取会员状态失败', error)
    } finally {
      loading.value = false
    }
  }
  
  // 检查是否可以使用某个功能
  const canUseFeature = (feature: keyof MembershipBenefits): boolean => {
    if (!benefits.value) return false
    
    // 如果是高级统计，需要会员
    if (feature === 'advanced_stats' && !benefits.value.advanced_stats) {
      return false
    }
    
    // 如果是AI预测，需要会员
    if (feature === 'ai_predict' && !benefits.value.ai_predict) {
      return false
    }
    
    // 如果是导出数据，需要会员
    if (feature === 'export_data' && !benefits.value.export_data) {
      return false
    }
    
    return true
  }
  
  // 检查每日限制
  const checkDailyLimit = (): boolean => {
    if (!benefits.value) return false
    const todayUsed = status.value?.today_used || 0

// app/api/cron/route.ts
import { NextRequest, NextResponse } from 'next/server'
import clientPromise from '@/lib/mongodb'

// 验证定时任务密钥
const validateCronRequest = (req: NextRequest) => {
  const authHeader = req.headers.get('authorization')
  const expectedToken = `Bearer ${process.env.CRON_SECRET}`
  return authHeader === expectedToken
}

export async function GET(req: NextRequest) {
  // 验证请求
  if (!validateCronRequest(req)) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }
  
  try {
    // 1. 获取最新开奖数据
    const latestData = await fetchLatestLotteryData()
    
    // 2. 存入数据库
    const client = await clientPromise
    const db = client.db()
    
    // 检查是否已存在
    const existing = await db.collection('lottery_results').findOne({
      lottery_type: 'kuaile8',
      draw_no: latestData.draw_no
    })
    
    if (!existing) {
      await db.collection('lottery_results').insertOne({
        lottery_type: 'kuaile8',
        draw_no: latestData.draw_no,
        draw_date: new Date(),
        numbers: parseNumbers(latestData.number),
        created_at: new Date()
      })
    }
    
    // 3. 清理旧数据（保留最近2年）
    const twoYearsAgo = new Date()
    twoYearsAgo.setFullYear(twoYearsAgo.getFullYear() - 2)
    await db.collection('lottery_results').deleteMany({
      lottery_type: 'kuaile8',
      draw_date: { $lt: twoYearsAgo }
    })
    
    return NextResponse.json({ success: true, data: latestData })
  } catch (error) {
    console.error('Cron job failed:', error)
    return NextResponse.json({ error: 'Internal error' }, { status: 500 })
  }
}
